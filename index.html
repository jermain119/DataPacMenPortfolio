SQL Projects

Project 1: State Orders Count


SELECT
  state,
  COUNT(DISTINCT order_id) AS num_orders
FROM
  `your-project-name.warehouse_orders.orders` AS orders
JOIN
  `your-project-name.warehouse_orders.warehouse` AS warehouse
ON orders.warehouse_id = warehouse.warehouse_id
GROUP BY
  warehouse.state;

Project 2: Average Sales


SELECT
  product_id,
  AVG(sale_amount) AS avg_sales
FROM
  `your-project-name.sales_data.transactions`
GROUP BY
  product_id;

Project 3: Top Customers


SELECT
  customer_id,
  SUM(purchase_amount) AS total_spent
FROM
  `your-project-name.sales_data.purchases`
GROUP BY
  customer_id
ORDER BY
  total_spent DESC
LIMIT 10;

Project 4: Product Sales by Region


SELECT
  region,
  product_id,
  SUM(sales) AS total_sales
FROM
  `your-project-name.sales_data.sales_by_region`
GROUP BY
  region, product_id;

Project 5: Monthly Revenue


SELECT
  DATE_TRUNC(month, sale_date) AS sale_month,
  SUM(revenue) AS total_revenue
FROM
  `your-project-name.revenue_data.sales`
GROUP BY
  sale_month
ORDER BY
  sale_month;
